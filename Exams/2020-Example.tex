\question Pergunta 11

$R = 2 Mbit/s = 2 \cdot 10^{6} bits/s$

$T_{p} = 250 ms = 0.250 s$   

$L = 250 Bytes = 2000 bits$

$BER = 10^{-4}$

$
FER = 1 - (1 - BER)^{L} 
        = 1 - (1 - 10^{-4})^{2000}
        = 0.181277
$

$
T_{f} = \frac{L}{R}
      = \frac{2000}{2 \cdot 10^{6}}
      = 0.001
$

$
a = \frac{T_{p}}{T_{f}}
  = \frac{0.250}{0.001}
  = 250
$

$
W_{max} = \frac{M}{2} 
        = \frac{64}{2}
        = 32
$

$
S = \frac{W \cdot (1 - FER)}{1 + 2a}
  = \frac{32 \cdot (1 - 0.181277)}{1 + 2 \cdot 250}
  = 0.052294
  = 5.1 \%
$

$
R_{max} = S \cdot R
        = 0.052294 \cdot 2 \cdot 10^{6}
        = 104587 bits/s
        = 105 kbits/s
$

\question Pergunta 13

$
L' = 1.3 \cdot L
   = 1.3 \cdot 2000
   = 2600
$

$FER' = 5\%$

$
T_{f}' = \frac{L}{R}
       = \frac{2600}{2 \cdot 10^{6}}
       = 0.0013
$

$
a' = \frac{T_{p}}{T_{f}'}
   = \frac{0.250}{0.0013}
   = 192.308
$

$
S' = \frac{W \cdot (1 - FER')}{1 + 2a'}
   = \frac{32 \cdot (1 - 0.05)}{1 + 2 \cdot 192.308}
   = 0.078835
$

$
R_{max}' = S' \cdot R
         = 0.078835 \cdot 2 \cdot 10^{6}
         = 157670 bits/s
         = 158 kbits/s
$

\question Pergunta 14

$\lambda = 600$

$L = 1500 \text{ bytes} = 12000 \text{ bits}$

$\rho = 100\% - 40\% = 0.6$

$
N = \frac{\rho}{1 - \rho}
  = \frac{0.6}{1 - 0.6}
  = 1.5
$

$
T = \frac{N}{\lambda} 
  = \frac{1.5}{600}
  = 0.0025 \text{ s}
  = 2.5 \text{ ms}
$
